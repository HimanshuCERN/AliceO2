46 void EveWorkflowHelper::draw(std::string jsonPath, int numberOfFiles, int numberOfTracks)
47 {
48  size_t nTracks = mTrackSet.trackGID.size();
49  if (numberOfTracks != -1 && numberOfTracks < nTracks) {
50  nTracks = numberOfTracks; // less than available
51  }
52  for (size_t it = 0; it < nTracks; it++) {
53  const auto& gid = mTrackSet.trackGID[it];
54  auto tim = mTrackSet.trackTime[it];
55 
56  switch (gid.getSource()) {
57  case GID::TPC:
58  drawTPC(gid, tim);
59  break;
60  case GID::MFT:
61  drawMFT(gid, tim);
62  break;
63  case GID::MCH:
64  drawMCH(gid, tim);
65  break;
66  case GID::ITS:
67  drawITS(gid, tim);
68  break;
69  case GID::ITSTPCTOF:
70  drawITSTPCTOF(gid, tim);
71  break;
72  case GID::ITSTPC:
73  drawITSTPC(gid, tim);
74  break;
75  default:
76  LOG(INFO) << "Track type " << gid.getSource() << " not handled";
77  }
78  }
79  FileProducer producer(jsonPath, numberOfFiles);
80  mEvent.toFile(producer.newFileName());
81 }

83 std::vector<PNT> EveWorkflowHelper::getTrackPoints(const o2::track::TrackPar& trc, float minR, float maxR, float maxStep, float minZ, float maxZ)
84 {
85  // adjust minR according to real track start from track starting point
86  float rMin = std::sqrt(trc.getX() * trc.getX() + trc.getY() * trc.getY());
87  if (rMin > minR) {
88  minR = rMin;
89  }
90  // prepare space points from the track param
91  std::vector<PNT> pnts;
92  int nSteps = std::max(2, int((maxR - minR) / maxStep));
93  const auto prop = o2::base::Propagator::Instance();
94  float xMin = trc.getX(), xMax = maxR * maxR - trc.getY() * trc.getY();
95  if (xMax > 0) {
96  xMax = std::sqrt(xMax);
97  }
98 
99  float dx = (xMax - xMin) / nSteps;
100  auto tp = trc;
101  float dxmin = std::abs(xMin - tp.getX()), dxmax = std::abs(xMax - tp.getX());
102 
103  if (dxmin > dxmax) { //start from closest end
104  std::swap(xMin, xMax);
105  dx = -dx;
106  }
107  if (!prop->propagateTo(tp, xMin, false, 0.99, maxStep, o2::base::PropagatorF::MatCorrType::USEMatCorrNONE)) {
108  return pnts;
109  }
110  auto xyz = tp.getXYZGlo();
111  pnts.emplace_back(PNT{xyz.X(), xyz.Y(), xyz.Z()});
112  for (int is = 0; is < nSteps; is++) {
113  if (!prop->propagateTo(tp, tp.getX() + dx, false, 0.99, 999., o2::base::PropagatorF::MatCorrType::USEMatCorrNONE)) {
114  return pnts;
115  }
116  xyz = tp.getXYZGlo();
117  if (xyz.Z() < minZ) {
118  return pnts;
119  }
120  if (xyz.Z() > maxZ) {
121  return pnts;
122  }
123  pnts.emplace_back(PNT{xyz.X(), xyz.Y(), xyz.Z()});
124  }
125  return pnts;
126 }

205 // TPC cluseters for given TPC track (gid)
206 void EveWorkflowHelper::drawTPCClusters(GID gid, float trackTime)
207 {
208  const auto& trc = mRecoCont.getTPCTrack(gid);
209  auto mTPCTracksClusIdx = mRecoCont.getTPCTracksClusterRefs();
210  auto mTPCClusterIdxStruct = &mRecoCont.getTPCClusters();
211  const auto& elParam = o2::tpc::ParameterElectronics::Instance();
212 
213  float clusterTimeBinOffset = trc.getTime0(); // in in time beans time assigned to track - primary vertex
214 
215  // FIXME: THIS MUST NOT BE CREATED EVERY TIME
216  // IT SHOULD BE CREATED IN THE INITIALIZATION, AND LATER WE NEED TO UPDATE IT REGULARLY
217  static std::unique_ptr<gpu::TPCFastTransform> fastTransform = (o2::tpc::TPCFastTransformHelperO2::instance()->create(0));
218 
219  // store the TPC cluster positions
220  for (int iCl = trc.getNClusterReferences(); iCl--;) {
221  uint8_t sector, row;
222  const auto& clTPC = trc.getCluster(mTPCTracksClusIdx, iCl, *mTPCClusterIdxStruct, sector, row);
223 
224  std::array<float, 3> xyz;
225  fastTransform->TransformIdeal(sector, row, clTPC.getPad(), clTPC.getTime(), xyz[0], xyz[1], xyz[2], clusterTimeBinOffset); // in sector coordinate
226  o2::math_utils::rotateZ(xyz, o2::math_utils::sector2Angle(sector % o2::tpc::SECTORSPERSIDE)); // lab coordinate (global)
227  mEvent.addCluster(xyz[0], xyz[1], xyz[2], trackTime);
228  }
229 }
